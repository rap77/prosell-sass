**Brain #7 Condition B — Auth pattern for integration tests (mandatory):**
Integration tests MUST use `app.dependency_overrides[get_current_auth_user_from_cookie]` to inject
the test user. The `db_session` is real (no override). This keeps test isolation clean and avoids
relying on HTTP cookies or JWT generation in tests.

Pattern to use in conftest.py:
```python
from prosell.infrastructure.api.dependencies.auth import get_current_auth_user_from_cookie

@pytest_asyncio.fixture
async def async_client_as_admin(app, admin_user):
    """AsyncClient with admin user injected via dependency_override."""
    app.dependency_overrides[get_current_auth_user_from_cookie] = lambda: admin_user
    async with AsyncClient(app=app, base_url="http://test") as client:
        yield client
    app.dependency_overrides.clear()

@pytest_asyncio.fixture
async def async_client_as_seller(app, seller_user):
    """AsyncClient with seller user injected via dependency_override."""
    app.dependency_overrides[get_current_auth_user_from_cookie] = lambda: seller_user
    async with AsyncClient(app=app, base_url="http://test") as client:
        yield client
    app.dependency_overrides.clear()
```

`db_session` must be a REAL session (no mock) to test actual DB interactions.
Role assignment for users MUST be done via `user.roles = [Role(...)]` — do NOT use `role=` string arg
unless a helper function builds the Role entity correctly.

"""Integration tests — Category API (Phase 12, Plans 12-01).

Tests: SC-1 (attribute_schema in create/response) and SC-2 (role-based active filtering).
Uses real DB with rollback. No repository mocks.
"""

import pytest
import pytest_asyncio
from uuid import uuid4
from httpx import AsyncClient


# ─── SC-1: POST /categories returns attribute_schema ──────────────────────

@pytest.mark.asyncio
async def test_create_category_returns_attribute_schema(async_client: AsyncClient, admin_user):
    """POST /categories returns 201 with attribute_schema in response."""
    schema = {
        "color": {"type": "string", "required": True},
        "year": {"type": "number", "required": False},
    }
    payload = {
        "name": f"Cars-{uuid4().hex[:8]}",
        "slug": f"cars-{uuid4().hex[:8]}",
        "tenant_id": str(admin_user.tenant_id),
        "attribute_schema": schema,
    }
    response = await async_client.post("/api/v1/categories", json=payload)

    assert response.status_code == 201
    data = response.json()
    assert "attribute_schema" in data
    assert data["attribute_schema"]["color"]["type"] == "string"


@pytest.mark.asyncio
async def test_create_category_without_attribute_schema_defaults_empty(
    async_client: AsyncClient, admin_user
):
    """POST /categories without attribute_schema defaults to {}."""
    payload = {
        "name": f"NoSchema-{uuid4().hex[:8]}",
        "slug": f"no-schema-{uuid4().hex[:8]}",
        "tenant_id": str(admin_user.tenant_id),
    }
    response = await async_client.post("/api/v1/categories", json=payload)

    assert response.status_code == 201
    data = response.json()
    assert data["attribute_schema"] == {}


# ─── SC-2: Role-based active filtering ────────────────────────────────────

@pytest.mark.asyncio
async def test_admin_sees_inactive_categories(async_client: AsyncClient, admin_user):
    """Admin user can see is_active=False categories."""
    # Create category then deactivate it
    slug = f"inactive-{uuid4().hex[:8]}"
    create_resp = await async_client.post("/api/v1/categories", json={
        "name": f"Inactive-{uuid4().hex[:8]}",
        "slug": slug,
        "tenant_id": str(admin_user.tenant_id),
        "is_active": False,
    })
    assert create_resp.status_code == 201
    cat_id = create_resp.json()["id"]

    # Admin should see it
    list_resp = await async_client.get("/api/v1/categories")
    assert list_resp.status_code == 200
    ids = [c["id"] for c in list_resp.json()["categories"]]
    assert cat_id in ids


@pytest.mark.asyncio
async def test_seller_does_not_see_inactive_categories(
    async_client_as_admin: AsyncClient,
    async_client_as_seller: AsyncClient,
    admin_user,
):
    """Seller user cannot see is_active=False categories.

    Criteria verificable (Brain #7 Condition D fix):
    1. Admin creates a category with is_active=False
    2. Admin can see it in list (is_admin=True → no filter)
    3. Seller gets list → the inactive category ID is NOT in the response
    """
    # Step 1: Admin creates inactive category
    slug = f"hidden-{uuid4().hex[:8]}"
    create_resp = await async_client_as_admin.post(
        "/api/v1/categories",
        json={
            "name": f"Hidden-{uuid4().hex[:8]}",
            "slug": slug,
            "tenant_id": str(admin_user.tenant_id),
            "is_active": False,
        },
    )
    assert create_resp.status_code == 201
    cat_id = create_resp.json()["id"]

    # Step 2: Admin CAN see it (baseline verification)
    admin_list = await async_client_as_admin.get("/api/v1/categories")
    assert admin_list.status_code == 200
    admin_ids = [c["id"] for c in admin_list.json()["categories"]]
    assert cat_id in admin_ids, "Admin should see inactive category"

    # Step 3: Seller CANNOT see it — inactive categories are filtered out
    seller_list = await async_client_as_seller.get("/api/v1/categories")
    assert seller_list.status_code == 200
    seller_ids = [c["id"] for c in seller_list.json()["categories"]]
    assert cat_id not in seller_ids, "Seller must NOT see inactive categories"


# ─── PATCH /categories/{id}/attribute-schema ──────────────────────────────

@pytest.mark.asyncio
async def test_update_attribute_schema_replaces_existing(
    async_client: AsyncClient, admin_user
):
    """PATCH /categories/{id}/attribute-schema replaces schema (REPLACE semantics)."""
    # Create category with initial schema
    create_resp = await async_client.post("/api/v1/categories", json={
        "name": f"Schema-{uuid4().hex[:8]}",
        "slug": f"schema-{uuid4().hex[:8]}",
        "tenant_id": str(admin_user.tenant_id),
        "attribute_schema": {"old_field": {"type": "string"}},
    })
    assert create_resp.status_code == 201
    cat_id = create_resp.json()["id"]

    # Replace schema
    new_schema = {"new_field": {"type": "number", "required": True}}
    patch_resp = await async_client.patch(
        f"/api/v1/categories/{cat_id}/attribute-schema",
        json={"attribute_schema": new_schema},
    )
    assert patch_resp.status_code == 200
    data = patch_resp.json()
    assert "old_field" not in data["attribute_schema"]
    assert "new_field" in data["attribute_schema"]


# ─── DELETE /categories/{id} ──────────────────────────────────────────────

@pytest.mark.asyncio
async def test_delete_category_soft_deletes(async_client: AsyncClient, admin_user):
    """DELETE /categories/{id} soft-deletes (deactivates) the category."""
    create_resp = await async_client.post("/api/v1/categories", json={
        "name": f"ToDelete-{uuid4().hex[:8]}",
        "slug": f"to-delete-{uuid4().hex[:8]}",
        "tenant_id": str(admin_user.tenant_id),
    })
    assert create_resp.status_code == 201
    cat_id = create_resp.json()["id"]

    delete_resp = await async_client.delete(f"/api/v1/categories/{cat_id}")
    assert delete_resp.status_code == 204

    # GET should return is_active=False (admin sees it, but deactivated)
    get_resp = await async_client.get(f"/api/v1/categories/{cat_id}")
    assert get_resp.status_code == 200
    assert get_resp.json()["is_active"] is False
