    - Checkbox column for selection
    - `overscan: 10` for smooth scrolling

    **Implementation skeleton:**
    ```tsx
    'use client'

    import { useRef, useMemo } from 'react'
    import { useReactTable, getCoreRowModel, getSortedRowModel } from '@tanstack/react-table'
    import { useVirtualizer } from '@tanstack/react-virtual'
    import { useDataGrid } from '@/lib/hooks/useDataGrid'
    import { DataGridRow } from './DataGridRow'

    export function DataGrid({ data }: { data: Vehicle[] }) {
      const {
        sorting,
        setSorting,
        rowSelection,
        setRowSelection,
      } = useDataGrid()

      // Stable columns (prevent re-renders)
      const columns = useMemo(() => [
        // Column definitions here
      ], [])

      const table = useReactTable({
        data,
        columns,
        state: { sorting, rowSelection },
        onSortingChange: setSorting,
        onRowSelectionChange: setRowSelection,
        getCoreRowModel: getCoreRowModel(),
        getSortedRowModel: getSortedRowModel(),
      })

      // Virtualization
      const tableContainerRef = useRef<HTMLDivElement>(null)
      const rowVirtualizer = useVirtualizer({
        count: table.getRowModel().rows.length,
        getScrollElement: () => tableContainerRef.current,
        estimateSize: () => 60, // Row height
        overscan: 10,
      })

      const virtualRows = rowVirtualizer.getVirtualItems()

      return (
        <div ref={tableContainerRef} className="h-[600px] overflow-auto">
          <table style={{ height: `${rowVirtualizer.getTotalSize()}px` }}>
            <thead> {/* Header */} </thead>
            <tbody>
              {virtualRows.map(virtualRow => {
                const row = table.getRowModel().rows[virtualRow.index]
                return (
                  <DataGridRow key={row.id} row={row} />
                )
              })}
            </tbody>
          </table>
        </div>
      )
    }
    ```

    - Only ~40 rows rendered (20 visible + 10 buffer top + 10 buffer bottom)
    - `useMemo` for columns definition
    - `memo()` for DataGridRow component

  <name>Task 7: Create vehicles API client with TanStack Query</name>
  <files>apps/web/src/lib/api/vehicles.ts</files>
  <action>
    Create API client for vehicle CRUD operations using TanStack Query:

    **Endpoints:**
    - `GET /api/v1/vehicles` - List vehicles with filters (tenant_id from JWT)
    - `GET /api/v1/vehicles/:id` - Get single vehicle
    - `POST /api/v1/vehicles` - Create vehicle
    - `PATCH /api/v1/vehicles/:id` - Update vehicle
    - `DELETE /api/v1/vehicles/:id` - Delete vehicle

    **Implementation:**
    ```tsx
    import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'

    interface Vehicle {
      id: string
      title: string
      price: number
      status: string
      // ... other fields
    }

    export function useVehicles(filters?: { status?: string }) {
      return useQuery({
        queryKey: ['vehicles', filters],
        queryFn: async () => {
          const res = await fetch(`/api/v1/vehicles?${new URLSearchParams(filters)}`)
          if (!res.ok) throw new Error('Failed to fetch vehicles')
          return res.json()
        },
        staleTime: 60 * 1000, // 1 minute
      })
    }

    export function useUpdateVehicle() {
      const queryClient = useQueryClient()

      return useMutation({
        mutationFn: ({ id, ...data }: Vehicle) =>
          fetch(`/api/v1/vehicles/${id}`, {
            method: 'PATCH',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(data),
          }).then(res => res.json()),

        onMutate: async (newVehicle) => {
          await queryClient.cancelQueries({ queryKey: ['vehicles'] })
          const previousVehicles = queryClient.getQueryData(['vehicles'])
          queryClient.setQueryData(['vehicles'], (old: Vehicle[]) =>
            old?.map(v => v.id === newVehicle.id ? newVehicle : v)
          )
          return { previousVehicles }
        },

        onError: (err, newVehicle, context) => {
          queryClient.setQueryData(['vehicles'], context.previousVehicles)
        },

        onSettled: () => {
          queryClient.invalidateQueries({ queryKey: ['vehicles'] })
        },
      })
    }
    ```

    **Security:**
    - All requests include httpOnly cookies (auth handled by browser)
    - Backend validates tenant_id from JWT (not from request)
    - Error handling with toast notifications (sonner)

    Create `apps/web/src/lib/api/` directory if it doesn't exist.

    Note: Backend API endpoints will be implemented in Phase 2 (Catalog & Roles). For now, use mock data or placeholder endpoints.
