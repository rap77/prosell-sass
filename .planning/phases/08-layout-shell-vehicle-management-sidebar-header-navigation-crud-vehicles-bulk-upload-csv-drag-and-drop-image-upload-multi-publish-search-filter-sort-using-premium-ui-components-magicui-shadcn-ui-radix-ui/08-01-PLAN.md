    ```typescript
    export async function middleware(request: NextRequest) {
      const token = request.cookies.get('access_token')?.value

      // 1. Auth check
      if (!token) {
        return NextResponse.redirect(new URL('/login', request.url))
      }

      const user = await verifyToken(token) // Use existing AuthProvider logic
      const { pathname } = request.url

      // 2. Role-based route protection
      if (pathname.startsWith('/admin') && user.role !== 'admin') {
        return NextResponse.redirect(new URL('/dashboard', request.url))
      }

      if (pathname.startsWith('/dealer') && user.role !== 'dealer') {
        return NextResponse.redirect(new URL('/dashboard', request.url))
      }

      if (pathname.startsWith('/manager') && user.role !== 'manager') {
        return NextResponse.redirect(new URL('/dashboard', request.url))
      }

      // 3. Smart redirect for /dashboard
      if (pathname === '/dashboard') {
        const roleHome = {
          admin: '/admin/dashboard',
          seller: '/catalog',
          dealer: '/dealer/reports',
          manager: '/manager/team',
        }[user.role]

        return NextResponse.redirect(new URL(roleHome, request.url))
      }

      return NextResponse.next()
    }

    export const config = {
      matcher: ['/((?!api|_next/static|_next/image|favicon.ico).*)'],
    }
    ```

    **Security constraints:**
    - ALL routes protected except /login, /register, /public
    - Tenant validation: Check pathname includes /tenant/{tenantId} matches JWT tenant_id
    - Zero latency: Stateless JWT validation (no DB calls in middleware)

    Note: Reuse token verification logic from AuthProvider if possible.
